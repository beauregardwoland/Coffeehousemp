	id = naval_professionalism.1

	desc = "naval_professionalism_1_desc"

		any_owned_province = {
			has_port= yes
		}

		}
	}
}

country_event = { ##good
	id = naval_professionalism.3
	title = "naval_professionalism_3"
	desc = "naval_professionalism_3_desc"
	picture = ECONOMY_eventPicture
	trigger = {
		is_at_war = no
		any_owned_province = {
			has_port = yes
		}
	}
	is_triggered_only = yes
	option = {
		name = "naval_professionalism_3_optiona"
		if = {
			limit = { army_professionalism = 1 }
			add_army_tradition = 10 
		}
		else = {
			add_army_professionalism = 0.05
		}
	}
	option = {
		name = "naval_professionalism_3_optionb"
		if = {
			limit = { 
				has_government_power = {
					mechanic_type = naval_prof_mechanic
					power_type = naval_prof
					value = 100
				}
			}
			add_navy_tradition = 10 
		}
		else = {
			add_government_power = {
				mechanic_type = naval_prof_mechanic
				power_type = naval_prof
				value = 5
			}
		}
	}
	option = {
		name = "naval_professionalism_3_optionc"
		add_years_of_income = 0.5
	}
}
