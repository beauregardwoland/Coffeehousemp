custom_button = {
    name = toggle_event_logi
    potential = {
		ai = no
	}
	trigger = {
	}
	effect = {
		if = {
			limit = { has_country_flag = gib_event }
			clr_country_flag = gib_event
		}
		else = {
			set_country_flag = gib_event
		}
	}
	tooltip = toggle_event_logi_tt
}
