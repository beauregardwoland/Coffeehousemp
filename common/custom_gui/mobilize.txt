custom_button = {
    name = mobilize_buttin
    potential = {
		#has_country_modifier = enabled_mobilization
	}
	trigger = {
		has_country_modifier = enabled_mobilization
		custom_trigger_tooltip = {
			tooltip = mobing_event_lock
			NOT = { has_country_flag = mobing_question }
		}
		NOT = { has_country_modifier = demobilization }
		NOT = { has_country_modifier = demobilization_bad }
	}
	effect = {
		if = {
			limit = { 
				or = {
					has_country_modifier = "mobilization_lite"
					has_country_modifier = "mobilization_scraping"
					has_country_modifier = "mobilization"
				}
			}
			country_event = {
				id = beaus_event_mobilization.2
			}
			set_country_flag = mobing_question
		}
		else = {
			country_event = {
				id = beaus_event_mobilization.1
			}
			add_years_of_income = -0.70
			set_country_flag = mobing_question
		}
		
	}
	tooltip = province_sel
}
