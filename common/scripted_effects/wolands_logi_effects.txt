update_goods = {
	set_variable = { which = $GOODS$_produced value = 0.00 }          	
	every_owned_province = { 
		limit = { 
			trade_goods = $GOODS$
		}

		export_to_variable = {
			variable_name = $GOODS$_produced    
			value = modifier:trade_goods_size                         
		}	
		
		export_to_variable = {
			variable_name = tradegood_modifier     
			value = modifier:trade_goods_size_modifier                        
		}
		
		export_to_variable = {
			variable_name = tradegood_modifier_glob    
			value = modifier:global_trade_goods_size_modifier 
			who = owner 
		}
		
		change_variable = {            
			which = tradegood_modifier 
			which = tradegood_modifier_glob  
		}

		multiply_variable = {
			which = tradegood_modifier
			which = $GOODS$_produced
		}

		change_variable = {            
			which = $GOODS$_produced
			which = tradegood_modifier 
		}
		
		owner = {
			change_variable = {            
				which = $GOODS$_produced
				which = PREV  
			}
		}
	}
}
beau_raw_goods = {
	set_variable = {
		which = food_trade_good
		value = 0
	}
	set_variable = {
		which = metals_trade_good
		value = 0
	}
	change_variable = {
		which = food_trade_good
		which = livestock_produced
	}
	change_variable = {
		which = food_trade_good
		which = fish_produced
	}
	change_variable = {
		which = food_trade_good
		which = grain_produced
	}
	if = {
		limit = { current_age = age_of_discovery }
		multiply_variable = {
			which = food_trade_good
			value = 10
		}
	}
	else_if = {
		limit = { current_age = age_of_reformation }
		multiply_variable = {
			which = food_trade_good
			value = 10
		}
	}
	else_if = {
		limit = { current_age = age_of_absolutism }
		multiply_variable = {
			which = food_trade_good
			value = 10
		}
	}
	else = {
		multiply_variable = {
			which = food_trade_good
			value = 10
		}
	}
	
	change_variable = {
		which = metals_trade_good
		which = iron_produced
	}
	change_variable = {
		which = metals_trade_good
		which = copper_produced
	}
	if = {
		limit = { current_age = age_of_discovery }
		multiply_variable = {
			which = metals_trade_good

	}
	else_if = {

			which = metals_trade_good

		}
	}
	else_if = {

			which = metals_trade_good

		}
	}
	else = {

			which = metals_trade_good
			value = 10
		}
	}
}
